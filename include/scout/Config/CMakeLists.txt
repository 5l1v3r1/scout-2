# Notes
# 
#  We use CMake's configure_file() feature to auto-generate build-time
#  configuration settings for inclusion into the Scout build.  This 
#  includes features like search paths, link-time libraries, and basic
#  flags on settings (e.g. is CUDA enabled).  
#  
#####

project(sc_config_headers)

  # Bugs exist in 2.8.8 so we'll insist folks update...
  cmake_minimum_required(VERSION 2.8.9) 

  ##### CONFIGURATION FILE GENERATION
  #
  set(SC_CONFIGURE_INC_DIR ${SC_BUILD_DIR}/scout-config/include/scout/Config)

  message(STATUS "scout: generating defs.h for build-time configuration settings.")
  message(STATUS "       location: ${SC_CONFIGURE_INC_DIR}/defs.h")

  configure_file (
    defs.h.cmake 
    ${SC_CONFIGURE_INC_DIR}/defs.h)

  #
  #####
